### Задача 1.2: Вектор в 3D (task01_2.c)
```C
#include <stdio.h>
#include <math.h>
#include <string.h>

#define MAX_NAME 50

struct Vector3D {
    double x, y, z;
    char name[MAX_NAME];
};

double dot_product(struct Vector3D a, struct Vector3D b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

struct Vector3D cross_product(struct Vector3D a, struct Vector3D b) {
    struct Vector3D result;
    result.x = a.y * b.z - a.z * b.y;
    result.y = a.z * b.x - a.x * b.z;
    result.z = a.x * b.y - a.y * b.x;
    strcpy(result.name, "cross");
    return result;
}

double vector_length(struct Vector3D v) {
    return sqrt(v.x * v.x + v.y * v.y + v.z * v.z);
}

void print_vector(struct Vector3D v) {
    printf("%s = (%.2f, %.2f, %.2f)\n", v.name, v.x, v.y, v.z);
}

int main() {
    struct Vector3D v1 = {1.0, 2.0, 3.0, "v1"};
    struct Vector3D v2 = {4.0, 5.0, 6.0, "v2"};
    
    print_vector(v1);
    print_vector(v2);
    
    double dot = dot_product(v1, v2);
    printf("Dot product: %.2f\n", dot);
    
    struct Vector3D cross = cross_product(v1, v2);
    print_vector(cross);
    
    printf("Length of v1: %.2f\n", vector_length(v1));
    printf("Length of v2: %.2f\n", vector_length(v2));
    
    return 0;
}
```

### Задача 1.3: Комплексная экспонента (task01_3.c)
```C
#include <stdio.h>
#include <math.h>

#define EPSILON 1e-10
#define MAX_ITER 100

struct Complex {
    double re;
    double im;
};

struct Complex c_add(struct Complex a, struct Complex b) {
    struct Complex result;
    result.re = a.re + b.re;
    result.im = a.im + b.im;
    return result;
}

struct Complex c_mul(struct Complex a, struct Complex b) {
    struct Complex result;
    result.re = a.re * b.re - a.im * b.im;
    result.im = a.re * b.im + a.im * b.re;
    return result;
}

struct Complex c_div_scalar(struct Complex a, double scalar) {
    struct Complex result;
    result.re = a.re / scalar;
    result.im = a.im / scalar;
    return result;
}

void print_complex(struct Complex z, const char* name) {
    if (z.im >= 0)
        printf("%s = %.6f + %.6fi\n", name, z.re, z.im);
    else
        printf("%s = %.6f - %.6fi\n", name, z.re, -z.im);
}

struct Complex c_exp(struct Complex z) {
    struct Complex result = {1.0, 0.0};
    struct Complex term = {1.0, 0.0};
    struct Complex z_pow = {1.0, 0.0};
    
    for (int n = 1; n <= MAX_ITER; n++) {
        z_pow = c_mul(z_pow, z);
        
        term = c_div_scalar(z_pow, (double)n);
        
        if (fabs(term.re) < EPSILON && fabs(term.im) < EPSILON) {
            break;
        }
        
        result = c_add(result, term);
    }
    
    return result;
}

int main() {
    struct Complex z = {0.5, 0.5};
    struct Complex exp_z = c_exp(z);
    
    print_complex(z, "z");
    print_complex(exp_z, "exp(z)");
    
    struct Complex zero = {0.0, 0.0};
    struct Complex exp_zero = c_exp(zero);
    print_complex(exp_zero, "exp(0)");
    
    return 0;
}
```
### Задача 1.4: Битовые поля для даты (task01_4.c)
```C
#include <stdio.h>

struct Date {
    unsigned int day : 5;    // 0-31 (5 бит)
    unsigned int month : 4;  // 1-12 (4 бита)
    unsigned int year : 12;  // 0-4095 (12 бит)
};

int is_valid_date(struct Date d) {
    if (d.day < 1 || d.day > 31) return 0;
    if (d.month < 1 || d.month > 12) return 0;
    if (d.year > 4095) return 0;
    
    int days_in_month[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    if (d.day > days_in_month[d.month]) return 0;
    
    return 1;
}

void print_date(struct Date d) {
    printf("%02d.%02d.%04d\n", d.day, d.month, d.year);
}

int main() {
    struct Date d1 = {25, 12, 2024};
    struct Date d2 = {31, 2, 2024};
    
    printf("Date 1: ");
    print_date(d1);
    printf("Valid: %s\n", is_valid_date(d1) ? "Yes" : "No");
    
    printf("Date 2: ");
    print_date(d2);
    printf("Valid: %s\n", is_valid_date(d2) ? "Yes" : "No");
    
    printf("Size of Date: %lu bytes\n", sizeof(struct Date));
    
    return 0;
}
```

### Задача 1.5: Двусвязный список (task01_5.c)
```C
#include <stdio.h>
#include <stdlib.h>

struct Node {
    int data;
    struct Node* next;
    struct Node* prev;
};

struct Node* create_node(int data) {
    struct Node* new_node = (struct Node*)malloc(sizeof(struct Node));
    new_node->data = data;
    new_node->next = NULL;
    new_node->prev = NULL;
    return new_node;
}

void append(struct Node** head, struct Node** tail, int data) {
    struct Node* new_node = create_node(data);
    
    if (*head == NULL) {
        *head = new_node;
        *tail = new_node;
        return;
    }
    
    (*tail)->next = new_node;
    new_node->prev = *tail;
    *tail = new_node;
}

void print_forward(struct Node* head) {
    printf("Forward: ");
    struct Node* current = head;
    while (current != NULL) {
        printf("%d ", current->data);
        current = current->next;
    }
    printf("\n");
}

void print_backward(struct Node* tail) {
    printf("Backward: ");
    struct Node* current = tail;
    while (current != NULL) {
        printf("%d ", current->data);
        current = current->prev;
    }
    printf("\n");
}

void free_list(struct Node* head) {
    struct Node* current = head;
    while (current != NULL) {
        struct Node* temp = current;
        current = current->next;
        free(temp);
    }
}

int main() {
    struct Node* head = NULL;
    struct Node* tail = NULL;
    
    append(&head, &tail, 10);
    append(&head, &tail, 20);
    append(&head, &tail, 30);
    append(&head, &tail, 40);
    append(&head, &tail, 50);
    
    print_forward(head);
    print_backward(tail);
    
    free_list(head);
    return 0;
}
```

### Задача 2.4: Размеченное объединение (task02_4.c)
```C
#include <stdio.h>
#include <stdlib.h>

enum DataType {
    INT_TYPE,
    FLOAT_TYPE,
    CHAR_TYPE
};

union Data {
    int i;
    float f;
    char c;
};

struct TaggedData {
    enum DataType tag;
    union Data data;
};

void print_tagged_data(struct TaggedData item) {
    switch (item.tag) {
        case INT_TYPE:
            printf("INT: %d\n", item.data.i);
            break;
        case FLOAT_TYPE:
            printf("FLOAT: %.2f\n", item.data.f);
            break;
        case CHAR_TYPE:
            printf("CHAR: '%c'\n", item.data.c);
            break;
        default:
            printf("Unknown type\n");
    }
}

int main() {
    int n;
    printf("Enter number of elements: ");
    scanf("%d", &n);
    
    struct TaggedData* arr = (struct TaggedData*)malloc(n * sizeof(struct TaggedData));
    
    for (int i = 0; i < n; i++) {
        int type = i % 3;
        switch (type) {
            case 0:
                arr[i].tag = INT_TYPE;
                arr[i].data.i = i * 10;
                break;
            case 1:
                arr[i].tag = FLOAT_TYPE;
                arr[i].data.f = i * 1.5f;
                break;
            case 2:
                arr[i].tag = CHAR_TYPE;
                arr[i].data.c = 'A' + i;
                break;
        }
    }
    
    printf("\nArray contents:\n");
    for (int i = 0; i < n; i++) {
        print_tagged_data(arr[i]);
    }
    
    free(arr);
    return 0;
}
```
